function eliminarSeleccionados(selectDisponibles,selectSeleccionadas){
  if(!selectDisponibles||!selectSeleccionadas)return;
  var hashSeleccionadas={};
  for(var i=0;i&lt;selectSeleccionadas.options.length;i++){
    hashSeleccionadas[selectSeleccionadas.options[i].value]=selectSeleccionadas.options[i];
  }
  for(var i=0;i&lt;selectDisponibles.options.length;){
    if(hashSeleccionadas[selectDisponibles.options[i].value]!=undefined){
      try{
        selectDisponibles.options[i].parentNode.removeChild(selectDisponibles.options[i]);
      }catch(e){}
    }else{
      i++;
    }
  }
}

          // Comprobar si la tecla pulsada es alguna de las teclas especiales
          // (teclas de borrado y flechas horizontales)
          var tecla_especial = false;
          for(var i in teclas_especiales) {
            if(codigoCaracter == teclas_especiales[i]  &amp;&amp; sncaracter=="S") {
              tecla_especial = true;
              break;
            }
          }
         
          // Comprobar si la tecla pulsada se encuentra en los caracteres permitidos
          // o si es una tecla especial
          return permitidos.indexOf(caracter) != -1 || tecla_especial;
}

    function agregarFila(table){
        var tbl             = document.getElementById(table);        
        var idTrBase        = $("#"+table+" tr")[1].id;
        var idTrNuevo       = "tr"+(Number($("#"+table+" tr:last").attr('id').substr(2)) + 1);
        var tr              = tbl.getElementsByTagName("tr")[1];
        var opcionesObjArr  = tr.getElementsByTagName("td");
        var newRow          = tbl.insertRow(tbl.rows.length);       
        
        newRow.setAttribute('id', idTrNuevo);
        for (var i=0; i&lt;opcionesObjArr.length; i++) {
            var oldCell = opcionesObjArr[i];
            var texto   = oldCell.innerHTML;
            var newCell = newRow.insertCell(i);
            newCell.innerHTML = texto;
            var inputActual = newCell.children[0];
            if(inputActual.type == 'text') {
                inputActual.value = '';
            }else if(inputActual.type == 'select-one') {
                inputActual.id = replaceAll(inputActual.id, idTrBase.substr(2),  idTrNuevo.substr(2));  
                inputActual.value = '';
                inputActual.selectedIndex = 0;
                if(inputActual.getAttribute('onchange') != null &amp;&amp; inputActual.getAttribute('onchange') != 'undefined') {
                    var txtOnChange = inputActual.getAttribute('onchange').toString();
                    if( txtOnChange.indexOf('{')&gt;0){
                        txtOnChange = txtOnChange.substr(Number(txtOnChange.indexOf('{')+1));
                        txtOnChange = txtOnChange.substr(0,Number(txtOnChange.indexOf('}')-1));
                    }
                    txtOnChange = replaceAll(txtOnChange, idTrBase.substr(2), idTrNuevo.substr(2));
                    var onchange = new Function(txtOnChange);
                    //document.datosClienteRootForm.elements[inputActual.id].onchange = onchange;
                    inputActual.onchange = onchange;
                }
            }else if(inputActual.tagName == 'A') {
                inputActual.style.display = '';
                inputActual.href = replaceAll(inputActual.href, idTrBase.substr(2), idTrNuevo.substr(2));
            }            
        }                
    }
    
    function replaceAll( str, from, to ) {
    var idx = str.indexOf( from );
    while ( idx &gt; -1 ) {
        str = str.replace( from, to ); 
        idx = str.indexOf( from );
    }
    return str;
    }
